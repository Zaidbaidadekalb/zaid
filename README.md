local function QHVF_fake_script() -- OK.LocalScript 
	local script = Instance.new('LocalScript', OK)

	script.Parent.MouseButton1Click:Connect(function()
		local plr = game.Players.LocalPlayer
	    local char = plr.Character
	    local hum = char.Humanoid
		hum.WalkSpeed = script.Parent.Parent.TextBox.Text
	end)
end
coroutine.wrap(QHVF_fake_script)()
local function PNIJG_fake_script() -- Frame.LocalScript 
	local script = Instance.new('LocalScript', Frame)

	function PopupGui()
		if script.Parent.Visible==true then script.Parent.Visible=false
		else script.Parent.Visible=true
		end
	end
	
	
	game:GetService("UserInputService").InputBegan:Connect(function(key)
		if key.KeyCode==Enum.KeyCode.K then
			PopupGui()
		end
	end)
end
coroutine.wrap(PNIJG_fake_script)()
